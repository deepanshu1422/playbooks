Dijkstra's Algorithm

const dijkstra = (graph, start) => {
  const distances = {};
  const visited = new Set();
  const pq = new PriorityQueue();
  
  // Initialize distances
  for (let node in graph) {
    distances[node] = node === start ? 0 : Infinity;
  }
  
  pq.enqueue(start, 0);
  
  while (!pq.isEmpty()) {
    const current = pq.dequeue();
    
    if (visited.has(current)) continue;
    visited.add(current);
    
    for (let neighbor of graph[current]) {
      const distance = distances[current] + neighbor.weight;
      
      if (distance < distances[neighbor.node]) {
        distances[neighbor.node] = distance;
        pq.enqueue(neighbor.node, distance);
      }
    }
  }
  
  return distances;
};

type Store struct {
    data  map[string]*Entry
    mu    sync.RWMutex
    ttl   *TTLManager
}

type Entry struct {
    Value      interface{}
    Type       DataType
    ExpiresAt  time.Time
}

func (s *Store) Get(key string) (interface{}, error) {
    s.mu.RLock()
    defer s.mu.RUnlock()
    
    entry, exists := s.data[key]
    if !exists {
        return nil, ErrKeyNotFound
    }
    
    if s.isExpired(entry) {
        go s.Delete(key) // Lazy deletion
        return nil, ErrKeyNotFound
    }
    
    return entry.Value, nil
}

func (s *Store) Set(key string, value interface{}, ttl time.Duration) error {
    s.mu.Lock()
    defer s.mu.Unlock()
    
    entry := &Entry{
        Value: value,
        Type:  TypeString,
    }
    
    if ttl > 0 {
        entry.ExpiresAt = time.Now().Add(ttl)
        s.ttl.Register(key, entry.ExpiresAt)
    }
    
    s.data[key] = entry
    s.replicateCommand("SET", key, value, ttl)
    
    return nil
}

2. Consistent Hashing

type ConsistentHash struct {
    ring        map[uint32]string
    sortedKeys  []uint32
    nodes       map[string]bool
    replicas    int
    mu          sync.RWMutex
}

func (ch *ConsistentHash) AddNode(node string) {
    ch.mu.Lock()
    defer ch.mu.Unlock()
    
    for i := 0; i < ch.replicas; i++ {
        hash := ch.hash(fmt.Sprintf("%s:%d", node, i))
        ch.ring[hash] = node
        ch.sortedKeys = append(ch.sortedKeys, hash)
    }
    
    sort.Slice(ch.sortedKeys, func(i, j int) bool {
        return ch.sortedKeys[i] < ch.sortedKeys[j]
    })
    
    ch.nodes[node] = true
}

func (ch *ConsistentHash) GetNode(key string) string {
    ch.mu.RLock()
    defer ch.mu.RUnlock()
    
    hash := ch.hash(key)
    
    // Binary search for the appropriate node
    idx := sort.Search(len(ch.sortedKeys), func(i int) bool {
        return ch.sortedKeys[i] >= hash
    })
    
    if idx == len(ch.sortedKeys) {
        idx = 0
    }
    
    return ch.ring[ch.sortedKeys[idx]]
}

func (r *Replication) Replicate(cmd Command) error {

        wg.Add(1)
        go func(addr string) {
            defer wg.Done()
            r.sendToReplica(addr, cmd)
        }(replica)
    }
    
    wg.Wait()
    return nil
}

func (r *Replication) Sync() error {
    if r.role != RoleReplica {
        return nil
    }
    
    // Connect to master
    conn, err := net.Dial("tcp", r.master)
    if err != nil {
        return err
    }
    defer conn.Close()
    
    // Request sync
    fmt.Fprintf(conn, "PSYNC %d\r\n", r.offset)
    
    // Receive and apply commands
    reader := bufio.NewReader(conn)
    for {
        cmd, err := r.parseCommand(reader)
        if err != nil {
            return err
        }
        r.applyCommand(cmd)
        r.offset++
    }
}

type AOF struct {
    file      *os.File
    mu        sync.Mutex
    buffer    *bufio.Writer
    syncMode  SyncMode
}

func (aof *AOF) Append(cmd Command) error {
    aof.mu.Lock()
    defer aof.mu.Unlock()
    
    // Serialize command
    data := aof.serialize(cmd)
    
    // Write to buffer
    if _, err := aof.buffer.Write(data); err != nil {
        return err
    }
    
    // Sync based on mode
    switch aof.syncMode {
    case SyncAlways:
        return aof.buffer.Flush()
    case SyncEverySecond:
        // Periodic flush in background
        return nil
    case SyncNo:
        // OS decides when to sync
        return nil
    }
    
    return nil
}

func (aof *AOF) Replay() error {
    file, err := os.Open(aof.filename)
    if err != nil {
        return err
    }
    defer file.Close()
    
    scanner := bufio.NewScanner(file)
    for scanner.Scan() {
        cmd, err := aof.deserialize(scanner.Bytes())
        if err != nil {
            log.Printf("Warning: skipping corrupt command: %v", err)
            continue
        }
        aof.executor.Execute(cmd)
    }
    
    return scanner.Err()
}
